### 2.4 errors.ts - 에러 처리 (41KB)

```typescript
// 에러 유형별 처리 전략
export function handleAPIError(error: unknown): ErrorHandling {
  if (error instanceof Anthropic.APIError) {
    switch (error.status) {
      case 400:
        // Bad Request: 잘못된 파라미터
        return { strategy: 'fail', message: error.message }

      case 401:
        // Unauthorized: API 키 문제
        return { strategy: 'prompt_reauth', message: '인증이 필요합니다. /login 실행' }

      case 429:
        // Rate Limited: 요청 한도 초과
        return { strategy: 'retry_after', retryAfter: error.headers['retry-after'] }

      case 529:
        // Overloaded: API 과부하
        return { strategy: 'exponential_backoff', baseDelay: 1000 }

      case 500:
      case 502:
      case 503:
        // Server Error: 일시적 서버 오류
        return { strategy: 'retry', maxRetries: 3 }
    }
  }

  if (error instanceof Anthropic.APIConnectionError) {
    return { strategy: 'retry', maxRetries: 3, message: '연결 오류' }
  }

  return { strategy: 'fail', message: '알 수 없는 오류' }
}
```

### 2.5 withRetry.ts - 재시도 로직 (28KB)

```typescript
export async function withRetry<T>(
  operation: () => Promise<T>,
  options: RetryOptions = {}
): Promise<T> {
  const {
    maxRetries = 3,
    baseDelay = 1000,
    maxDelay = 30000,
    backoffFactor = 2,
  } = options

  let lastError: Error
  let delay = baseDelay

  for (let attempt = 0; attempt <= maxRetries; attempt++) {
    try {
      return await operation()
    } catch (error) {
      lastError = error as Error
      const handling = handleAPIError(error)

      if (handling.strategy === 'fail') throw error
      if (attempt === maxRetries) throw error

      // 지수 백오프
      if (handling.retryAfter) {
        delay = parseInt(handling.retryAfter) * 1000
      } else {
        delay = Math.min(delay * backoffFactor, maxDelay)
      }

      await sleep(delay + Math.random() * 1000)  // jitter 추가
    }
  }

  throw lastError!
}
```

### 3.2 MCP 클라이언트 (client.ts)

```typescript
export class MCPClient {
  name: string
  status: 'connected' | 'connecting' | 'disconnected' | 'error'
  tools: MCPTool[]
  resources: MCPResource[]
  private client: Client  // @modelcontextprotocol/sdk 클라이언트

  async connect(config: MCPServerConfig): Promise<void> {
    // 전송 방식에 따라 연결
    const transport = config.type === 'stdio'
      ? new StdioClientTransport({ command: config.command, args: config.args })
      : new SSEClientTransport({ url: config.url })

    await this.client.connect(transport)

    // 도구와 리소스 목록 가져오기
    const { tools } = await this.client.listTools()
    const { resources } = await this.client.listResources()

    this.tools = tools
    this.resources = resources
    this.status = 'connected'
  }

  async callTool(name: string, input: unknown): Promise<MCPToolResult> {
    return await this.client.callTool({ name, arguments: input })
  }
}
```

### 5.2 OAuth 플로우

```typescript
export async function startOAuthFlow(): Promise<string> {
  // 1. PKCE 코드 생성
  const { codeVerifier, codeChallenge } = generatePKCE()

  // 2. 인증 URL 생성
  const authUrl = new URL('https://claude.ai/oauth/authorize')
  authUrl.searchParams.set('client_id', CLAUDE_CODE_CLIENT_ID)
  authUrl.searchParams.set('response_type', 'code')
  authUrl.searchParams.set('code_challenge', codeChallenge)
  authUrl.searchParams.set('code_challenge_method', 'S256')
  authUrl.searchParams.set('redirect_uri', 'http://localhost:54321/callback')
  authUrl.searchParams.set('scope', 'openid profile')

  // 3. 브라우저 열기
  await openBrowser(authUrl.toString())

  // 4. 로컬 서버에서 콜백 대기
  const code = await waitForCallback(54321)

  // 5. 코드를 토큰으로 교환
  const tokens = await exchangeCodeForTokens(code, codeVerifier)

  // 6. 키체인에 저장
  await saveToKeychain('claude-code-token', tokens.access_token)

  return tokens.access_token
}
```

### 6.2 플러그인 로드

```typescript
export async function loadPlugins(): Promise<Plugin[]> {
  const pluginDirs = [
    path.join(os.homedir(), '.claude', 'plugins'),  // 글로벌
    path.join(getCwd(), '.claude', 'plugins'),       // 프로젝트
  ]

  const plugins: Plugin[] = []

  for (const dir of pluginDirs) {
    if (!existsSync(dir)) continue

    const entries = readdirSync(dir)
    for (const entry of entries) {
      const pluginPath = path.join(dir, entry, 'index.js')
      if (existsSync(pluginPath)) {
        const plugin = await import(pluginPath)
        plugins.push(plugin.default)
      }
    }
  }

  return plugins
}
```
